    ```typescript
    /**
     * E2E: Content Export Actions (SHIP-01 — Phase 24 download/redirect coverage)
     *
     * Verifies ExportActionsBar behavior:
     *   1. Download .txt — triggers browser download with correct filename
     *   2. Open in LinkedIn — opens popup with LinkedIn shareArticle URL
     *   3. Open in X — opens popup with twitter intent/tweet URL
     *   4. Instagram info — shows tooltip (no popup, copy-paste flow)
     *   5. Export bar visible — container renders when final_content exists
     *
     * All API calls mocked via page.route(). No real network calls.
     * Uses page.waitForEvent('download') and page.waitForEvent('popup') — no waitForTimeout.
     */

    import { test, expect } from "@playwright/test";
    import { mockExportContent } from "./helpers/mock-api";

    const JOB_ID = "export-test-job-001";
    const CONTENT_URL = `/dashboard/content/${JOB_ID}`;

    test.describe("Export Actions Bar", () => {

      test.beforeEach(async ({ page }) => {
        await mockExportContent(page, { jobId: JOB_ID, platform: "linkedin" });
      });

      test("export-actions-bar container is visible when final_content exists", async ({ page }) => {
        await page.goto(CONTENT_URL);
        await expect(page.locator('[data-testid="export-actions-bar"]')).toBeVisible();
      });

      test("clicking Download .txt triggers a file download", async ({ page }) => {
        await page.goto(CONTENT_URL);
        await page.locator('[data-testid="export-actions-bar"]').waitFor({ state: "visible" });

        const [download] = await Promise.all([
          page.waitForEvent("download"),
          page.locator('[data-testid="download-txt-btn"]').click(),
        ]);

        // Filename should match: linkedin-YYYY-MM-DD.txt
        const filename = download.suggestedFilename();
        expect(filename).toMatch(/^linkedin-\d{4}-\d{2}-\d{2}\.txt$/);
      });

      test("clicking Open in LinkedIn opens a new tab with LinkedIn shareArticle URL", async ({ page, context }) => {
        await page.goto(CONTENT_URL);
        await page.locator('[data-testid="export-actions-bar"]').waitFor({ state: "visible" });

        const [popup] = await Promise.all([
          context.waitForEvent("page"),
          page.locator('[data-testid="open-linkedin-btn"]').click(),
        ]);

        await popup.waitForLoadState("domcontentloaded");
        expect(popup.url()).toMatch(/linkedin\.com\/shareArticle/);
      });

      test("clicking Open in X opens a new tab with twitter intent URL", async ({ page, context }) => {
        await page.goto(CONTENT_URL);
        await page.locator('[data-testid="export-actions-bar"]').waitFor({ state: "visible" });

        const [popup] = await Promise.all([
          context.waitForEvent("page"),
          page.locator('[data-testid="open-x-btn"]').click(),
        ]);

        await popup.waitForLoadState("domcontentloaded");
        expect(popup.url()).toMatch(/twitter\.com\/intent\/tweet/);
      });

      test("clicking Instagram info shows tooltip (no popup)", async ({ page, context }) => {
        await page.goto(CONTENT_URL);
        await page.locator('[data-testid="export-actions-bar"]').waitFor({ state: "visible" });

        // Listen for a new popup — there should be none
        let popupOpened = false;
        context.on("page", () => { popupOpened = true; });

        await page.locator('[data-testid="instagram-info-btn"]').click();

        // Give 500ms for any accidental popup to open
        await page.waitForTimeout(500);
        expect(popupOpened).toBe(false);

        // Tooltip text should be visible
        await expect(page.locator("text=Instagram")).toBeVisible();
      });

    });
    ```
