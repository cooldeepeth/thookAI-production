```typescript
/**
 * Production Smoke Test — PERF-06 / PERF-08
 *
 * Hits the REAL production URL. No page.route() mocks.
 * Requires:
 *   PROD_URL=https://thook.ai (or Railway preview URL)
 *   SMOKE_USER_EMAIL=smoke-test@example.com (dedicated test account)
 *   SMOKE_USER_PASSWORD=<password>
 *
 * Run:
 *   PROD_URL=https://thook.ai \
 *   SMOKE_USER_EMAIL=... \
 *   SMOKE_USER_PASSWORD=... \
 *   npx playwright test e2e/production-smoke.spec.ts --project=chromium
 *
 * Cross-browser:
 *   npx playwright test e2e/production-smoke.spec.ts
 */

import { test, expect } from "@playwright/test";

const BASE_URL = process.env.PROD_URL || "http://localhost:3000";
const SMOKE_EMAIL = process.env.SMOKE_USER_EMAIL || "";
const SMOKE_PASSWORD = process.env.SMOKE_USER_PASSWORD || "";

test.use({ baseURL: BASE_URL });

test.beforeAll(async () => {
  if (!process.env.PROD_URL) {
    console.warn(
      "[production-smoke] PROD_URL not set — running against localhost:3000"
    );
  }
  if (!SMOKE_EMAIL || !SMOKE_PASSWORD) {
    throw new Error(
      "SMOKE_USER_EMAIL and SMOKE_USER_PASSWORD must be set for production smoke test"
    );
  }
});

test("smoke-01: landing page loads with key elements", async ({ page }) => {
  await page.goto("/");
  await expect(page).toHaveTitle(/ThookAI|Thook/i);
  // Hero section or Get Started CTA must be present
  const cta = page.getByRole("link", { name: /get started|sign up|try free/i }).first();
  await expect(cta).toBeVisible({ timeout: 10000 });
});

test("smoke-02: auth page renders and accepts credentials", async ({ page }) => {
  await page.goto("/auth");
  // Login form must be visible
  const emailInput = page.locator('input[type="email"]');
  await expect(emailInput).toBeVisible({ timeout: 8000 });

  // Attempt login with smoke credentials
  await emailInput.fill(SMOKE_EMAIL);
  const passInput = page.locator('input[type="password"]');
  await passInput.fill(SMOKE_PASSWORD);

  const submitBtn = page.locator('button[type="submit"]').first();
  await submitBtn.click();

  // After login, should redirect away from /auth (to /dashboard or onboarding)
  await expect(page).not.toHaveURL(/\/auth/, { timeout: 15000 });
  console.log(`[smoke-02] Redirected to: ${page.url()}`);
});

test("smoke-03: dashboard loads after auth", async ({ page }) => {
  // Navigate to auth and log in
  await page.goto("/auth");
  const emailInput = page.locator('input[type="email"]');
  await expect(emailInput).toBeVisible({ timeout: 8000 });
  await emailInput.fill(SMOKE_EMAIL);
  await page.locator('input[type="password"]').fill(SMOKE_PASSWORD);
  await page.locator('button[type="submit"]').first().click();

  // Wait for dashboard or onboarding
  await page.waitForURL(/(dashboard|onboarding)/, { timeout: 20000 });
  console.log(`[smoke-03] Post-login URL: ${page.url()}`);

  // Some dashboard element must be visible (sidebar nav, or onboarding wizard)
  const dashElement =
    page.getByTestId("sidebar-nav").or(
      page.getByTestId("onboarding-wizard").or(
        page.getByRole("navigation")
      )
    );
  await expect(dashElement.first()).toBeVisible({ timeout: 10000 });
});

test("smoke-04: health endpoint is healthy", async ({ request }) => {
  // Derive API URL from PROD_URL by replacing frontend domain with api subdomain
  const prodUrl = process.env.PROD_URL || "http://localhost:3000";
  const apiUrl = prodUrl.includes("thook.ai")
    ? prodUrl.replace("https://thook.ai", "https://api.thook.ai").replace("https://www.thook.ai", "https://api.thook.ai")
    : "http://localhost:8001";

  const apiBase = process.env.PROD_API_URL || apiUrl;
  console.log(`[smoke-04] API base: ${apiBase}`);

  const response = await request.get(`${apiBase}/health`);
  expect(response.status()).toBe(200);
  const body = await response.json();
  expect(body.status).toBe("ok");
});
```

- Requires `SMOKE_USER_EMAIL` and `SMOKE_USER_PASSWORD` — throws if missing
- Tests: landing loads, auth works, dashboard loads post-login, health endpoint healthy
