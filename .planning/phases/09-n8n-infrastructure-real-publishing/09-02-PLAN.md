    async def _verify_n8n_request(request: Request) -> dict:
        """Dependency that verifies n8n HMAC signature and parses body."""
        body_bytes = await request.body()

        if body_bytes:

    Before calling `_publish_to_platform`, atomically claim the post:
    ```python
    from datetime import datetime, timezone, timedelta

    claim = await db.scheduled_posts.find_one_and_update(
        {
            "schedule_id": post["schedule_id"],
            "status": "scheduled",
            # Only claim if not already being processed
            "$or": [
                {"processing_started_at": {"$exists": False}},
                {"processing_started_at": {"$lt": datetime.now(timezone.utc) - timedelta(minutes=5)}},
            ],
        },
        {"$set": {"status": "processing", "processing_started_at": datetime.now(timezone.utc)}},
        return_document=True,  # pymongo.ReturnDocument.AFTER equivalent for Motor
    )
    if not claim:
        logger.info("Post %s already being processed or claimed — skipping (idempotency guard)", post["schedule_id"])
        continue
    ```

    Also add a `last_published_at` check: if the post has `published_at` within the last 2 minutes, skip it:

    published_at = post.get("published_at")
    if published_at:

        if (datetime.now(timezone.utc) - published_at) < timedelta(minutes=2):
            logger.info("Post %s published within last 2 min — skipping (idempotency)", post["schedule_id"])
