import { render, screen, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { MemoryRouter } from 'react-router-dom';
import CookieConsent from '../../components/CookieConsent';

// Mock window.posthog to track init() calls
beforeEach(() => {
  localStorage.clear();
  window.posthog = {
    init: jest.fn(),
    opt_in_capturing: jest.fn(),
    opt_out_capturing: jest.fn(),
    __loaded: false,
    has_opted_out_capturing: () => false,
  };
});

test('consent_gate: posthog.init not called before accept', () => {
  render(<MemoryRouter><CookieConsent /></MemoryRouter>);
  // Banner should be visible (no stored consent)
  expect(window.posthog.init).not.toHaveBeenCalled();
});

test('consent_gate: posthog.init called after accept', async () => {
  render(<MemoryRouter><CookieConsent /></MemoryRouter>);
  const acceptBtn = screen.queryByText(/accept all/i) ||
                    screen.queryByTestId('consent-accept-btn');
  if (acceptBtn) {
    await userEvent.click(acceptBtn);
    expect(window.posthog.init).toHaveBeenCalledTimes(1);
    expect(localStorage.getItem('thookai_cookie_consent')).toBe('accepted');
  }
});

test('decline_gate: posthog.opt_out called on decline', async () => {
  render(<MemoryRouter><CookieConsent /></MemoryRouter>);
  const declineBtn = screen.queryByText(/essential only/i) ||
                     screen.queryByTestId('consent-decline-btn');
  if (declineBtn) {
    await userEvent.click(declineBtn);
    expect(window.posthog.opt_out_capturing).toHaveBeenCalledTimes(1);
    expect(localStorage.getItem('thookai_cookie_consent')).toBe('declined');
  }
});
