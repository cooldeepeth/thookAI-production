```python
try:
    redis_client = await get_redis()
    if redis_client:
        today = datetime.now(timezone.utc).strftime("%Y-%m-%d")
        rkey = f"gdpr_export:{user_id}:{today}"
        count = await redis_client.incr(rkey)
        if count == 1:
            await redis_client.expire(rkey, 86400)  # 24h TTL
        if count > 3:
            raise HTTPException(
                status_code=429,
                detail="Export limit reached (3 per day). Try again tomorrow."
            )
except HTTPException:
    raise
except Exception:
    pass  # Redis unavailable — allow export, log silently
```
