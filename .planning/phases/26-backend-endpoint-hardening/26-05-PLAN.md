```python
"""
Phase 26: Auth Guard Coverage Tests
Covers: BACK-01 (endpoints tested), BACK-04 (all protected endpoints reject unauthenticated)
Also validates BACK-03 (error_code in 401 responses — covered by Plan 02)

10 critical protected endpoints verified to return 401 with error_code=UNAUTHORIZED
when no session_token cookie is present.
"""
import pytest
from httpx import AsyncClient, ASGITransport


PROTECTED_ENDPOINTS = [
    # (method, path, description)
    ("GET", "/api/persona/me", "persona profile"),
    ("GET", "/api/dashboard", "dashboard stats"),
    ("GET", "/api/analytics/overview", "analytics overview"),
    ("GET", "/api/content/jobs", "content jobs list"),
    ("GET", "/api/billing/subscription", "billing subscription status"),
    ("GET", "/api/agency/workspaces", "agency workspaces"),
    ("GET", "/api/uploads/my-uploads", "user uploads"),
    ("GET", "/api/campaigns", "campaigns list"),
    ("GET", "/api/strategy/feed", "strategy recommendations"),
    ("GET", "/api/uom", "unit of measure"),
]


@pytest.mark.asyncio
async def test_all_critical_protected_endpoints_require_auth():
    """All 10 critical protected endpoints must return 401 with error_code when unauthenticated."""
    from server import app
    failures = []

    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        for method, path, description in PROTECTED_ENDPOINTS:
            if method == "GET":
                resp = await client.get(path)
            elif method == "POST":
                resp = await client.post(path, json={})
            else:
                continue

            if resp.status_code != 401:
                failures.append(
                    f"{method} {path} ({description}): expected 401, got {resp.status_code}"
                )
                continue

            data = resp.json()
            if "error_code" not in data:
                failures.append(
                    f"{method} {path} ({description}): 401 returned but no 'error_code' in body: {data}"
                )

    assert not failures, "Auth guard failures:\n" + "\n".join(failures)


@pytest.mark.asyncio
async def test_persona_me_requires_auth():
    """GET /api/persona/me: 401 + error_code (high sensitivity — user's voice fingerprint)"""
    from server import app
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        resp = await client.get("/api/persona/me")
    assert resp.status_code == 401
    assert resp.json().get("error_code") == "UNAUTHORIZED"
```
